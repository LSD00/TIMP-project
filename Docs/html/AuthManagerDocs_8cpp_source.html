147AuthResult AuthManager::verifyRegistrationCode(const QString& email, const QString& code, QString* errorMsg) {
148 QMutexLocker locker(&codesMutex);
149
150 if (!registrationCodes.contains(email)) return AuthResult::InvalidCode;
151
152 auto& ctx = registrationCodes[email];
153
154 if (QDateTime::currentDateTime() > ctx.expiresAt) {
155 registrationCodes.remove(email);
156 return AuthResult::CodeExpired;
157 }
158
159 if (ctx.code != code) {
160 ctx.attempts++;
161 if (ctx.attempts >= MAX_ATTEMPTS) {
162 registrationCodes.remove(email);
163 return AuthResult::TooManyAttempts;
164 }
165 return AuthResult::InvalidCode;
166 }
167
168 ctx.isVerified = true;
169
170 QMutexLocker dbLocker(&dbMutex);
171 QSqlQuery query(db);
172 query.prepare("INSERT INTO users (email) VALUES (:email) ON CONFLICT DO NOTHING");
173 query.bindValue(":email", email);
174 query.exec();
175
176 return AuthResult::Success;
177}

189bool AuthManager::sendEmailToUser(const QString& email, const QString& code) {
190 QSettings settings("config.ini", QSettings::IniFormat);
191 QString apiKey = settings.value("Email/api_key").toString();
192
193 if (apiKey.isEmpty()) {
194 qCritical() << "API Key is missing in config.ini!";
195 return false;
196 }
197
198 QNetworkAccessManager manager;
199 QNetworkRequest request(QUrl("https://api.brevo.com/v3/smtp/email"));
200 request.setHeader(QNetworkRequest::ContentTypeHeader, "application/json");
201 request.setRawHeader("api-key", apiKey.toUtf8());
202
203 QJsonObject body;
204 body["sender"] = QJsonObject{{"name", "AuthSystem"}, {"email", "ovcarenkoaleksandr180@gmail.com"}};
205 body["to"] = QJsonArray{QJsonObject{{"email", email}}};
206 body["subject"] = "Verification Code";
207 body["htmlContent"] = QString("Your code: <b>%1</b>").arg(code);
208
209 QEventLoop loop;
210 QNetworkReply* reply = manager.post(request, QJsonDocument(body).toJson());
211 connect(reply, &QNetworkReply::finished, &loop, &QEventLoop::quit);
212 loop.exec();
213
214 bool ok = (reply->error() == QNetworkReply::NoError);
215 qDebug() << "Sending email to:" << email << "code:" << code;
216
217 if (!ok) qWarning() << "Email error:" << reply->errorString();
218 qDebug() << "HTTP status:" << reply->attribute(QNetworkRequest::HttpStatusCodeAttribute).toInt();
219 qDebug() << "Response:" << reply->readAll();
220
221 reply->deleteLater();
222 return ok;
223}

234AuthResult AuthManager::authenticate(const QString& email, const QString& password) {
235 QMutexLocker locker(&dbMutex);
236 QSqlQuery query(db);
237 query.prepare("SELECT id FROM users WHERE email = :email "
238 "AND password_hash IS NOT NULL "
239 "AND password_hash = crypt(:pass, password_hash)");
240 query.bindValue(":email", email);
241 query.bindValue(":pass", password);
242
243 if (!query.exec()) return AuthResult::DatabaseError;
244 return query.next() ? AuthResult::Success : AuthResult::InvalidCredentials;
245}

305AuthResult AuthManager::finalizeRegistration(const QString& email, const QString& password, QString* errorMsg) {
306 QMutexLocker locker(&codesMutex);
307 if (!registrationCodes.contains(email) || !registrationCodes[email].isVerified)
308 return AuthResult::CodeNotVerified;
309
310 if (!validatePassword(password)) return AuthResult::WeakPassword;
311
312 QMutexLocker dbLocker(&dbMutex);
313 QSqlQuery query(db);
314 query.prepare("UPDATE users SET password_hash = crypt(:pass, gen_salt('bf')) WHERE email = :email");
315 query.bindValue(":pass", password);
316 query.bindValue(":email", email);
317
318 if (!query.exec()) return AuthResult::DatabaseError;
319
320 registrationCodes.remove(email);
321 return AuthResult::Success;
322}
