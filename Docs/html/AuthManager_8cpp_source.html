107AuthResult AuthManager::verifyRegistrationCode(const QString& email, const QString& code, QString* errorMsg) {
108 QMutexLocker locker(&codesMutex);
109
110 if (!registrationCodes.contains(email)) return AuthResult::InvalidCode;
111
112 auto& ctx = registrationCodes[email];
113
114 if (QDateTime::currentDateTime() > ctx.expiresAt) {
115 registrationCodes.remove(email);
116 return AuthResult::CodeExpired;
117 }
118
119 if (ctx.code != code) {
120 ctx.attempts++;
121 if (ctx.attempts >= MAX_ATTEMPTS) {
122 registrationCodes.remove(email);
123 return AuthResult::TooManyAttempts;
124 }
125 return AuthResult::InvalidCode;
126 }
127
128 ctx.isVerified = true;
129
130 // Вставляем пустую запись пользователя в базу для резервирования email
131 QMutexLocker dbLocker(&dbMutex);
132 QSqlQuery query(db);
133 query.prepare("INSERT INTO users (email) VALUES (:email) ON CONFLICT DO NOTHING");
134 query.bindValue(":email", email);
135 query.exec();
136
137 return AuthResult::Success;
138}

140void AuthManager::sendEmailToUserAsync(const QString& email, const QString& code) {
141 QSettings settings("config.ini", QSettings::IniFormat);

143

145 qCritical() << "API Key is missing in config.ini!";

150 QNetworkRequest request(QUrl("https://api.brevo.com/v3/smtp/email"));
151 request.setHeader(QNetworkRequest::ContentTypeHeader, "application/json");
152 request.setRawHeader("api-key", apiKey.toUtf8());
153
154 QJsonObject body;
155 body["sender"] = QJsonObject{{"name", "AuthSystem"}, {"email", "ovcarenkoaleksandr180@gmail.com"}};
156 body["to"] = QJsonArray{QJsonObject{{"email", email}}};
157 body["subject"] = "Verification Code";
158 body["htmlContent"] = QString("Your code: <b>%1</b>").arg(code);
159
160 QNetworkReply* reply = networkManager->post(request, QJsonDocument(body).toJson());
161
162 connect(reply, &QNetworkReply::finished, this, [reply, email]() {
163 if (reply->error() == QNetworkReply::NoError) {
164 qDebug() << "Email successfully sent to:" << email;
165 } else {
166 qWarning() << "Email error for" << email << ":" << reply->errorString();
167 qDebug() << "HTTP status:" << reply->attribute(QNetworkRequest::HttpStatusCodeAttribute).toInt();
168 qDebug() << "Response:" << reply->readAll();
169 }
170 reply->deleteLater();
171 });
172}

174AuthResult AuthManager::authenticate(const QString& email, const QString& password) {
175 QMutexLocker locker(&dbMutex);
176 QSqlQuery query(db);
177 // Проверка пароля напрямую в БД средствами pgcrypto
178 query.prepare("SELECT id FROM users WHERE email = :email "
179 "AND password_hash IS NOT NULL "
180 "AND password_hash = crypt(:pass, password_hash)");
181 query.bindValue(":email", email);
182 query.bindValue(":pass", password);
183
184 if (!query.exec()) return AuthResult::DatabaseError;
185 return query.next() ? AuthResult::Success : AuthResult::InvalidCredentials;
186}

213AuthResult AuthManager::finalizeRegistration(const QString& email, const QString& password, QString* errorMsg) {
214 QMutexLocker locker(&codesMutex);
215 if (!registrationCodes.contains(email) || !registrationCodes[email].isVerified)
216 return AuthResult::CodeNotVerified;
217
218 if (!validatePassword(password)) return AuthResult::WeakPassword;
219
220 QMutexLocker dbLocker(&dbMutex);
221 QSqlQuery query(db);
222 // Генерация соли и хеширование пароля через pgcrypto
223 query.prepare("UPDATE users SET password_hash = crypt(:pass, gen_salt('bf')) WHERE email = :email");
224 query.bindValue(":pass", password);
225 query.bindValue(":email", email);
226
227 if (!query.exec()) return AuthResult::DatabaseError;
228
229 registrationCodes.remove(email);
230 return AuthResult::Success;
231}
