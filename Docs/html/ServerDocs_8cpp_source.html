83void Server::handleRequest(QTcpSocket* client, const QJsonObject& request) {
84 QString action = request.value("action").toString();
85 QString email = request.value("email").toString();
86 QString errorMsg;
87
88 // ==========================================
89 // 1. ОТКРЫТЫЕ ЭНДПОИНТЫ (Не требуют логина)
90 // ==========================================
91
92 if (action == "login") {
93 AuthResult res = AuthManager::getInstance().authenticate(email, request.value("password").toString());
94 if (res == AuthResult::Success) {
95 authenticatedClients[client] = true; // Устанавливаем статус "Авторизован"
96 }
97 sendAuthResponse(client, res);
98 return;
99 }
100
101 // --- Цикл регистрации ---
102 if (action == "register_request") {
103 sendAuthResponse(client, AuthManager::getInstance().requestRegistrationCode(email, &errorMsg), errorMsg);
104 return;
105 }
106 if (action == "register_verify") {
107 sendAuthResponse(client, AuthManager::getInstance().verifyRegistrationCode(email, request.value("code").toString(), &errorMsg), errorMsg);
108 return;
109 }
110 if (action == "register_finalize") {
111 sendAuthResponse(client, AuthManager::getInstance().finalizeRegistration(email, request.value("password").toString(), &errorMsg), errorMsg);
112 return;
113 }
114
115 // --- Цикл сброса пароля ---
116 if (action == "reset_request") {
117 sendAuthResponse(client, AuthManager::getInstance().requestPasswordReset(email, &errorMsg), errorMsg);
118 return;
119 }
120 if (action == "reset_verify") {
121 sendAuthResponse(client, AuthManager::getInstance().verifyResetCode(email, request.value("code").toString(), &errorMsg), errorMsg);
122 return;
123 }
124 if (action == "reset_finalize") {
125 sendAuthResponse(client, AuthManager::getInstance().resetPassword(email, request.value("password").toString(), request.value("password_confirm").toString(), &errorMsg), errorMsg);
126 return;
127 }
128
129 // ==========================================
130 // 2. ЗАЩИЩЕННЫЕ ЭНДПОИНТЫ (Требуют логина)
131 // ==========================================
132
133 // Проверка прав доступа: если флаг сессии равен false, отсекаем запрос.
134 if (!authenticatedClients.value(client, false)) {
135 sendResponse(client, {{"status", "error"}, {"message", "Unauthorized"}});
136 return;
137 }
138
139 // Запрос на вычисление графиков
140 if (action == "calc") {
141 auto res = MainCalcSafe(
142 request.value("a").toInt(),
143 request.value("b").toInt(),
144 request.value("c").toInt()
145 );
146
147 QJsonArray arr;
148 for (const auto& p : res) {
149 // Предполагается, что p имеет поля x и y
150 arr.append(QJsonObject{{"x", p.x}, {"y", p.y}});
151 }
152
153 sendResponse(client, {{"status", "success"}, {"data", arr}});
154 return;
155 }
156
157 // Если экшен неизвестен
158 sendResponse(client, {{"status", "error"}, {"message", "Unknown action"}});
159}
